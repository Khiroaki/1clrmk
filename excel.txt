&НаСервере
Процедура ЗагрузкаДанныхИзФайлаНаСервере (ТабДок, ПутьКФайлу)
	ТабДок.Прочитать (ПутьКФайлу, СпособЧтенияЗначенийТабличногоДокумента.Значение);
	ПостроительЗапроса = Новый ПостроительЗапроса;
	ПостроительЗапроса.ИсточникДанных = Новый ОписаниеИсточникаДанных (ТабДок.Область("Люди"));
	ПостроительЗапроса.Выполнить();
	ТаблицаСотрудниковИзФайла = ПостроительЗапроса.Результат.Выгрузить();
	Для Каждого СтрокаСотрудника Из ТаблицаСотрудниковИзФайла Цикл 
		НовыйСотрудник = Справочники.Сотрудники.СоздатьЭлемент();
		Запрос = Новый Запрос;
		Запрос.Текст = "ВЫБРАТЬ
		               |	Организации.Название КАК Название
		               |ИЗ
		               |	Справочник.Организации КАК Организации
		               |ГДЕ
		               |	Организации.Название = &Название";
		Запрос.Параметры.Вставить("Название", СтрокаСотрудника.Организация);
		Организация = Запрос.Выполнить().Выгрузить();
		Если Организация.Количество() = 0 Тогда
			НоваяОрганизация = Справочники.Организации.СоздатьЭлемент();
			НоваяОрганизация.Наименование = СтрокаСотрудника.Организация;
			НоваяОрганизация.Название = СтрокаСотрудника.Организация;
			НоваяОрганизация.Записать();
			
			НовыйСотрудник.Организация = НоваяОрганизация;
		Иначе
			НовыйСотрудник.Организация = Организация[0].Название;
		КонецЕсли;
		НовыйСотрудник.Наименование = СтрокаСотрудника.Имя;
		НовыйСотрудник.Имя = СтрокаСотрудника.Имя;
	    НовыйСотрудник.ФИО = СтрокаСотрудника.ФИО;
		НовыйСотрудник.Записать();
	КонецЦикла;
КонецПроцедуры
